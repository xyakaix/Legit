$domainDN = ([ADSI]"LDAP://RootDSE").defaultNamingContext

# Try DomainDnsZones first
Write-Host "[*] Checking DomainDnsZones..." -ForegroundColor Yellow
try {
    $base = New-Object DirectoryServices.DirectoryEntry("LDAP://DC=DomainDnsZones,$domainDN")
    $searcher = New-Object DirectoryServices.DirectorySearcher($base)
    $searcher.Filter = "(objectClass=dnsZone)"
    $searcher.PropertiesToLoad.Add("dc") | Out-Null
    $searcher.FindAll() | ForEach-Object {
        Write-Host "  [+] Zone: $($_.Properties['dc'][0])" -ForegroundColor Green
        Write-Host "      Path: $($_.Path)" -ForegroundColor Cyan
    }
} catch {
    Write-Host "  [-] DomainDnsZones not found" -ForegroundColor Red
}
